  \begin{question}
   Suppose $F(x)=\p(\xi\leq x)$ is a probability distribution function. Define the following four functions:
   \begin{equation*}
    \begin{split}
     G_1(y)=&\sup\{x:F(x)< y\}\\
     G_2(y)=&\sup\{x:F(x)\leq y\}\\
     G_3(y)=&\inf\{x:F(x)> y\}\\
     G_4(y)=&\inf\{x:F(x)\geq y\}\\
    \end{split}
   \end{equation*}
   Prove the following property of $G_j(y),j=1,\cdots,4$.
   \begin{enumerate}[(1)]
    \item $G_1(y)=G_4(y)$ and $G_2(y)=G_3(y)$.
    \item $G_j(y),j=1,\cdots,4$ are all non-decreasing.
    \item $G_1(y)\leq G_2(y)$.
    \item $F(G_1(y) )=F(G_4(y) )\geq y$.
    \item $F\big(G_2(y)-0 \big)=F\big(G_3(y)-0 \big)\leq y$.
    \item $\p\big(G_1(y)<\xi<G_2(y) \big)=0$. In other words, if $G_1(y)<G_2(y)$, then $F(x)$ is flat in the interval $\big(G_1(y),G_2(y)\big)$.
    \item If $G_1(y)< G_2(y)$, then $F(G_1(y))=y$.
    \item If $\p(X=x)=F(x)-F(x-0)>0$, then $G_1(F(x))=x$.
    \item $x<G_1(y)\Rightarrow F(x)<y$ and $x>G_2(y)\Rightarrow F(x)>y$.
    \item $G_1(y)$ is left continuous and $G_2(y)$ is right continuous. 
   \end{enumerate}
  \end{question}

     \{x:F(x)< y\}\leq \{x:F(x)\geq y\},\quad \{x:F(x)\leq  y\}\leq\{x:F(x)> y\}
    \end{equation*}
    Then we know

     \begin{split}
      G_1(y)=&\sup\{x:F(x)< y\}\leq \inf\{x:F(x)\geq y\}=G_4(y)\\

    \end{equation*}

    If $G_1(y)<G_4(y)$, for all $z\in \big(G_1(y),G_4(y)\big)$, by the definition of supremum and infimum, then $z\notin \{x:F(x)< y\}\bigcup \{x:F(x)\geq y\}=\r$. This is a contradiction.
